double Bound(int layer,int cv,int cw){

    double sv = double(cv);
    int cr = capacity - cw;

    while (i<set_number && goods[i].weight <= cr)
    {

			// layerä»£è¡¨å›æº¯å±‚æ•°ï¼Œå…¶ä½™å‚æ•°åˆ†åˆ«ä»£è¡¨å½“å‰ä»·å€¼ä»¥åŠå½“å‰é‡é‡

				// æ³¨æ„è¿™ä¸ªåœ°æ–¹ï¼Œä¸ç”¨å¤§äº ï¼ï¼ï¼

				if (current_value > maxValue && current_weight <= capacity) {

					maxValue = current_value;

					for (int i = 0; i < set_number; i++) {
						max_set[i] = current_set[i];

					}
				}
			}
			else
			{
				for (int j = 1; j >= 0; j--) {

					current_set[layer] = j;

					current_weight += (goods[layer].weight * current_set[layer]);
					current_value += (goods[layer].value * current_set[layer]);

					// ç»§ç»­è¿›è¡Œå­æ ‘çš„é€’å½’
					brute_force(layer + 1, current_value, current_weight);

					// é€’å½’å‡ºæ¥åå›æ»šæ•°æ®
					current_weight -= (goods[layer].weight * current_set[layer]);
					current_value -= (goods[layer].value * current_set[layer]);

				}
			}

		}

class Bag_with_backtrack : public Bag {		// ç»§æ‰¿ Bag çš„ä¿æŠ¤ä»¥åŠå…¬ç”¨æˆå‘˜
	
	protected:
		int current_set[MAX] = { 0 };		// define current_set to save current set


		int Bound_sv(int layer, int cv) {

			int i = layer;
			int sv = cv;

			for (i; i < set_number; i++) {
				sv += goods[i].value;		// å‰©ä½™ç‰©å“éƒ½è£…ä¸‹æ—¶çš„æœ€å¤§ä»·å€¼
			}

			return sv;

		}
		
		double Bound(int layer,int cv,int cw){
			int i = layer;
			
			double sv = double(cv);
			int cr = capacity - cw;

			while (i<set_number && goods[i].weight <= cr)
			{
				cr -= goods[i].weight;
				sv = sv + goods[i].value;
				i++;
			}

			if (i < set_number)
				sv = sv + goods[i].ave_value * cr;

			return sv;
		}
		

		// Using back track method to get max value
		void BackTrack(int layer, int current_value, int current_weight) {
			// layerä»£è¡¨å›æº¯å±‚æ•°ï¼Œå…¶ä½™å‚æ•°åˆ†åˆ«ä»£è¡¨å½“å‰ä»·å€¼ä»¥åŠå½“å‰é‡é‡
			//double expect_sumValue = 0;		// è®°å½•æœŸæœ›ä»·å€¼æ€»å’Œ

			if (layer == set_number) {		// if layer bigger than number, over
				// æ³¨æ„è¿™ä¸ªåœ°æ–¹ï¼Œä¸ç”¨å¤§äº ï¼ï¼ï¼

				if (current_value > maxValue) {

					maxValue = current_value;

					for (int i = 0; i < set_number; i++) {
						max_set[i] = current_set[i];

					}
				}
			}
			else
			{
				for (int j = 0; j <= 1; j++) {

					current_set[layer] = j;

					/************************å‰ªæ**************************/
					

					if (
						current_weight + current_set[layer] * goods[layer].weight <= capacity
						&& Bound_sv(layer,current_value) >= maxValue && Bound(layer,current_value,current_weight) > maxValue
						)
					{
						// æ›´æ–°å½“å‰èƒŒåŒ…ç‰©å“çš„æ€»é‡é‡ï¼Œç›®å‰èƒŒåŒ…ç‰©å“çš„æ€»ä»·å€¼
						current_weight += (goods[layer].weight * current_set[layer]);
						current_value += (goods[layer].value * current_set[layer]);

						// ç»§ç»­è¿›è¡Œå­æ ‘çš„é€’å½’
						BackTrack(layer + 1, current_value, current_weight);

						// é€’å½’å‡ºæ¥åå›æ»šæ•°æ®
						current_weight -= (goods[layer].weight * current_set[layer]);
						current_value -= (goods[layer].value * current_set[layer]);

					}
				}
			}

		}


	public:

		}

			t1 = getSystemTime(); // å¼€å§‹è®¡æ—¶

class Bag_with_branchAndbound :public Bag {		// ä» Bag ç±»ä¸­æ´¾ç”Ÿ

	protected:
		
		int tree_node_id = 1;		// è§£ç©ºé—´æ ‘ç»“ç‚¹id 1å¼€å§‹
		
		int lower_bound = 0;		// ç›®æ ‡å‡½æ•°çš„ä¸‹ç•Œ

		// è´ªå¿ƒæ³•æ±‚èƒŒåŒ…å®¹é‡çš„ä¸‹ç•Œ
		int greed() {

			int temp[MAX] = { 0 };		// å®šä¹‰ä¸´æ—¶æ•°ç»„ï¼Œç”¨æ¥è¡¨ç¤ºè´§ç‰©å‘é‡ï¼Œå¹¶åˆå§‹åŒ–ä¸º0
			int cap = capacity;			// æš‚å­˜èƒŒåŒ…å®¹é‡
			int w = 0;					// å½“å‰èƒŒåŒ…å®¹é‡
			int updown = 0;				// ä¸‹ç•Œ
			
			// å¯¹èƒŒåŒ…ç‰©å“æŒ‰ç…§å•ä½ä»·å€¼è¿›è¡Œæ’åº
			sort_set();

			// æ±‚è§£
			for (int i = 0; i < set_number; i++) {

				// å¦‚æœè¶…é‡ï¼Œè·³å‡º
				if ((w+goods[i].weight) > cap) {
					continue;
				}

				temp[i] = 1;
				w += goods[i].weight;
				updown += (temp[i] * goods[i].value);
			}

			cout << "è§£çš„ä¸‹ç•Œæ˜¯ï¼š" << updown << endl;

			return updown;

		}


		// é™ç•Œå‡½æ•°æ±‚ä¸Šç•Œ
		void limit(ND &n)				        //è®¡ç®—åˆ†æç»“ç‚¹çš„ä¸Šç•Œ
		{
			// ä¸‹ä¸€ä¸ªè¦é€‰çš„ç‰©å“å°±æ˜¯ç¬¬layerå±‚ç‰©å“
			int w = n.current_weight;
			double v = n.current_value;
			int i = n.layer;

			while (i < set_number && w + goods[i].weight <= capacity) {
				w += goods[i].weight;
				v += goods[i].value;
				i++;
			}

			if (i < set_number) {  // è£…éƒ¨åˆ†ç‰©å“
				n.ub = v + (capacity - w)*goods[i].ave_value;
			}
			else
			{
				n.ub = v;
			}
		}

		// ä¼˜å…ˆé˜Ÿåˆ—è¿›é˜Ÿæ“ä½œ
		void EnQueue(ND n, priority_queue<ND> &q) {
			if (n.layer == set_number) {		// å·²ç»åˆ°è¾¾å¶å­ç»“ç‚¹

				if (n.current_value > maxValue) {		// æ›´æ–°maxValue
					maxValue = n.current_value;
					for (int i = 0; i < set_number; i++) {
						max_set[i] = n.cset[i];
					}
				}
			}
			else {
				q.push(n);			// éå¶å­ç»“ç‚¹è¿›é˜Ÿ
			}
		}

		// åˆ†æ”¯é™ç•Œæ³•
		void branchAndbound() {

			// é¦–å…ˆä½¿ç”¨è´ªå¿ƒæ³•æ±‚è§£æ­¤goodsé›†åˆçš„ç›®æ ‡å€¼ä¸‹ç•Œ
			lower_bound = greed();

			priority_queue<ND> q;		// ä¼˜å…ˆé˜Ÿåˆ—
			ND n0, n1, n2;				// å…ˆå®šä¹‰æ ¹ç»“ç‚¹ï¼Œå·¦å­©å­ç»“ç‚¹ï¼Œå³å­©å­ç»“ç‚¹

			/*åˆå§‹åŒ–æ ¹èŠ‚ç‚¹*/
			n0.id = tree_node_id++;		
			n0.layer = 0;
			n0.current_value = 0;
			n0.current_weight = 0;
			for (int i = 0; i < set_number; i++) {
				n0.cset[i] = 0;
			}
			limit(n0);					// æ±‚æ ¹ç»“ç‚¹ä¸Šç•Œ
			q.push(n0);					// æ ¹ç»“ç‚¹è¿›é˜Ÿ

			while (!q.empty())			// é˜Ÿä¸ç©ºå¾ªç¯
			{
				n0 = q.top();				// å–é˜Ÿå¤´
				q.pop();				// å‡ºé˜Ÿ
				if (n0.current_weight + goods[n0.layer].weight <= capacity && n0.ub >= lower_bound) {	// è¶…é‡å‰ªæ å°äºæœ€å°æœŸæœ›é‡é‡å‰ªæ

					// è®¾ç½®å·¦å­©å­ç»“ç‚¹
					n1.id = tree_node_id++;
					n1.layer = n0.layer + 1;
					n1.current_weight = n0.current_weight + goods[n0.layer].weight;
					n1.current_value = n0.current_value + goods[n0.layer].value;
					/*å¤åˆ¶è§£å‘é‡*/
					for (int i = 0; i < set_number; i++) {
						n1.cset[i] = n0.cset[i];
					}
					n1.cset[n1.layer-1] = 1;			// æ›´æ–°è§£å‘é‡
					limit(n1);						// æ±‚é™ç•Œå‡½æ•°ä¸Šç•Œ
					EnQueue(n1, q);					// å·¦å­©å­è¿›é˜Ÿ
				}

				// è®¾ç½®å³å­©å­ç»“ç‚¹
				n2.id = tree_node_id++;
				n2.layer = n0.layer + 1;
				n2.current_value = n0.current_value;
				n2.current_weight = n0.current_weight;

				// å¤åˆ¶è§£å‘é‡
				for (int i = 0; i < set_number; i++) {
					n2.cset[i] = n0.cset[i];
				}

				
				n2.cset[n2.layer-1] = 0;		 // æ›´æ–°è§£å‘é‡
				limit(n2);					// æ±‚ç•Œé™å‡½æ•°ä¸Šç•Œ
				
				if (n2.ub > maxValue && n2.ub >= lower_bound) {		// å‰ªæï¼ŒæœŸæœ›æœ€å¤§å€¼å°äºå·²æœ‰æœ€å¤§å€¼å°±å‰ªæ  å°äºæœ€å°æœŸæœ›é‡é‡å‰ªæ
					EnQueue(n2, q);
				}

			}
		}


	public:

		Bag_with_branchAndbound(int N) :Bag(N) {

			cout << "åˆ†æ”¯é™ç•Œæ³•å¯¹è±¡å®ä¾‹åŒ–..." << endl;
		
		}

		void get_maxValue_by_branchAndbound() {
			
			// åˆ†æ”¯é™ç•Œæ³•
			t1 = getSystemTime();		// å¼€å§‹è®¡æ—¶
			branchAndbound();
			t2 = getSystemTime();		// ç»“æŸè®¡æ—¶

			// è®¡ç®—å¹¶æ‰“å°æ—¶é—´
			sum_time = t2 - t1;
			cout << "ä½¿ç”¨åˆ†æ”¯é™ç•Œæ³•æ±‚è§£é—®é¢˜è§„æ¨¡ä¸º" << set_number << " çš„0/1èƒŒåŒ…é—®é¢˜æ‰€éœ€è¦çš„æ—¶é—´ä¸ºï¼š" << sum_time << "ms." << endl;
			
			// æ‰“å°é›†åˆ
			print_set();
		}
